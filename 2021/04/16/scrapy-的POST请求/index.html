scrapy构造POST和GET请求的方式

POST

scrapy.FormRequest

yeild scrapy.FormRequest(
    url='https://........', 
    method='post',  # 选择post
    dont_filter=True, # 关闭去重, 非必须
    formdata=data, # post 数据根据情况 选择是  dict类型: data / json字符串类型: json.dumps(data)
    callback=self.parse)

scrapy.Request

跟我们平时用get请求的方法一样

yield scrapy.Request(
        method='POST',# 选择post
        url='https://...', 
        body=json.dumps(data), # 注意这里的key: body 就是指你需要请求的post data, 同样post 数据根据情况 选择是  dict类型: data / json字符串类型: json.dumps(data)
        headers=self.headers, # dict 类型
        callback=self.parse)

GET

一般GET请求

yield scrapy.Request(url, 
                     headers=headers,
                     cookies=cookies,
                     callback=self.parse_article)

有params的GET请求

params = {
    'widen': '1',
    'tadrequire': 'true',
}
yield scrapy.Request(url='https://www....', 
                     method='GET',
                     body=json.dumps(params),

请求参数设置方法

GET请求 查询字符串

1. 使用FormRequest

start_urls = ['https://careers.tencent.com/tencentcareer/api/post/Query?']

def start_requests(self):
    for url in self.start_urls:
        params = {
            "timestamp": str(int(time.time() * 1000)),
            "countryId": "",
            "cityId": "",
            "bgIds": "",
            "productId": "",
            "categoryId": "",
            "parentCategoryId": "",
            "attrId": "",
            "keyword": "",
            "pageIndex": "1",
            "pageSize": "10",
            "language": "zh-cn",
            "area": "cn"
        }
        yield scrapy.FormRequest(url, 
                                method="GET", 
                                formdata=params)

2. 使用urlencode

from urllib.parse import urlencode

def start_requests(self):
    for url in self.start_urls:
        params = {
            "timestamp": str(int(time.time() * 1000)),
            "countryId": "",
            "cityId": "",
            "bgIds": "",
            "productId": "",
            "categoryId": "",
            "parentCategoryId": "",
            "attrId": "",
            "keyword": "",
            "pageIndex": "1",
            "pageSize": "10",
            "language": "zh-cn",
            "area": "cn"
        }
        url = url + urlencode(params)
        yield scrapy.Request(url,
                             method="GET")
